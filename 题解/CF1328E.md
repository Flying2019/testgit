考虑假如某个点在一条到根节点的链上，那么他的父亲（如果存在）一定也在这条链上。那么题目变成了询问某些点是不是在一条链上。

随便怎么写都可以。
```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define N 200010
using namespace std;
int nxt[N<<1],to[N<<1],head[N],cnt;
void add(int u,int v)
{
	nxt[++cnt]=head[u];
	to[cnt]=v;
	head[u]=cnt;
}
int fa[N],dep[N],siz[N],son[N],top[N];
void dfs1(int u,int pre)
{
	dep[u]=dep[pre]+1;
	fa[u]=pre;
	siz[u]=1;
	int sonsiz=0;
	for(int i=head[u];i;i=nxt[i])
	{
		int v=to[i];
		if(v==pre) continue;
		dfs1(v,u);
		siz[u]+=siz[v];
		if(siz[v]>sonsiz) sonsiz=siz[v],son[u]=v;
	}
}
void dfs2(int u,int topp)
{
	top[u]=topp;
	if(son[u]) dfs2(son[u],topp);
	for(int i=head[u];i;i=nxt[i])
	{
		int v=to[i];
		if(v!=fa[u] && v!=son[u])dfs2(v,v);
	}
}
int num[N],tot;
bool cmp(int x,int y){return dep[x]==dep[y]?x<y:dep[x]>dep[y];}
int lca(int x,int y)//y is root of x
{
	while(top[x]!=top[y])
	{
		if(dep[x]<dep[y]) swap(x,y);
		x=fa[top[x]];
	}
	return (dep[x]<dep[y])?x:y;
}
bool check()
{
	sort(num+1,num+tot+1,cmp);
	int u=num[1];
	for(int i=2;i<=tot;i++)
	if(lca(u,num[i])!=num[i]) return false;
	return true;
}
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<n;i++)
	{
		int u,v;
		scanf("%d%d",&u,&v);
		add(u,v),add(v,u);
	}
	dfs1(1,0);
	dfs2(1,1);
	for(int i=1;i<=m;i++)
	{
		scanf("%d",&tot);
		for(int i=1;i<=tot;i++)
		{
			int a;
			scanf("%d",&a);
			if(a==1) num[i]=1;
			else num[i]=fa[a];
		}
		puts(check()?"YES":"NO");
	}
	return 0;
}
```
