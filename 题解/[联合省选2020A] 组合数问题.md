多项式转下降幂。
$$\binom{n}{m}\times m^{\underline a}=\frac{n!}{m!(n-m)!}\cdot\frac{m!}{(m-a)!}$$

$$=\frac{n!}{(n-m)!(m-a)!}$$

$$=\binom{n-a}{m-a}\times n^{\underline a}$$

$$\left(\sum_{k=0}^n f(k) \times x^k \times \binom n k\right) \bmod p$$

$$=\left(\sum_{k=0}^n \sum{b_ik^{\underline i}} \times x^k \times \binom n k\right) \bmod p$$

$$=\left(\sum_{k=0}^n \sum{b_in^{\underline i}} \times x^k \times \binom {n-i} {k-i}\right) \bmod p$$

$$=\left(\sum_{k=0}^n x^k\times \sum{b_in^{\underline i}} \times \binom {n-i} {k-i}\right) \bmod p$$

$$=\left(\sum{b_i\sum_{k=0}^n x^k\times \binom {n-i} {k-i}n^{\underline i}}\right) \bmod p$$

$$=\left(\sum{b_i\sum_{k=0}^{n-i} x^{k+i}\times \binom {n-i} {k}n^{\underline i}}\right) \bmod p$$

$$=\left(\sum{b_i\times x^in^{\underline i}\sum_{k=0}^{n-i} x^k\times \binom {n-i} {k}}\right) \bmod p$$

$$=\left(\sum{b_i\times x^in^{\underline i}\times(x+1)^{n-i}}\right) \bmod p$$

$$x^k=∑_{i=0}^{k}x^{\underline i}\times S_2(k,i)$$

$$\Rightarrow b_i=∑_{j=i}^{m}a_j×S_2(j,i)$$
```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#define N 5010
#define ll long long
using namespace std;
ll mod;
ll s[N][N];
int a[N],b[N];
ll ksm(ll a,ll b=mod-2)
{
	ll ans=1;
	for(;b;b>>=1)
	{
		if(b&1) ans=ans*a%mod;
		a=a*a%mod;
	}
	return ans;
}
ll fac[N];
int main()
{
	int n,x,m;
	scanf("%d%d%lld%d",&n,&x,&mod,&m);
	fac[0]=1;
	for(int i=0;i<=m;i++) scanf("%d",&a[i]),fac[i]=fac[i-1]*i%mod;
	s[0][0]=1;
	for(int i=1;i<=m;i++)
		for(int j=1;j<=m;j++)
		s[i][j]=(s[i-1][j-1]+s[i-1][j]*j)%mod;
	for(int i=0;i<=m;i++)
	{
		b[i]=0;
		for(int j=i;j<=m;j++) b[i]=(b[i]+a[j]*s[j][i])%mod;
	}
	ll ans=0;
	for(int i=0;i<=m;i++)
	{
		ll res=1;
		for(int j=0;j<i;j++) res=res*(n-j)%mod;
		ans=(ans+b[i]*ksm(x,i)%mod*res%mod*ksm(x+1,n-i))%mod;
	}
	printf("%lld",ans);
	return 0;
}
```
