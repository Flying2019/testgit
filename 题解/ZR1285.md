#include<cstdio>
#include<iostream>
#include<algorithm>
#define M 20
#define N 1<<(M+3)
using namespace std;
int num[N],f[N][M+3],n,x;
int main()
{
	scanf("%d",&n);
	for(int i=1,a;i<=n;i++)
		scanf("%d",&a),num[a]^=1;
	for(int i=n;i>=0;i--) num[i]^=num[i+1];
	for(int i=n;i>=0;i--)
		for(int j=0;j<=M;j++)
		{
			int u=1<<j,v=1<<(j+1);
			f[i][j]=f[i+v][j]^num[i+v]^num[i+u];
		}
	for(int i=1;i<=n;i++)
	{
		bool t=true;
		for(int j=0;t && (1<<j)<=i;j++)
		{
			int s=0;
			for(int k=0;k<=n;k+=i+1)
			{
				int y=min(k+i,n),z=k+(((y-k+1+(1<<(j+1)))>>(j+1))<<(j+1));
				s^=f[z][j]^f[k][j]^num[z]^num[max(y+1,z-(1<<j))];
			}
			if(s) printf("Alice "),t=false;
		}
		if(t) printf("Bob ");
	}
}
