### [链接](https://loj.ac/problem/3303)
法一：考虑操作实际是链+a和链异或和，按照套路倒建trie树模拟进位即可，$O(n\log n)$

法二：对每一位单独算贡献，发现某一点的某一位能影响到的点是若干段等长的区间，直接差分是$O(n^2)$，但是考虑这些差分处的二进制下最后几位都是相同的，可以直接用一个数组表示。复杂度$O(n\log n)$。
```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#define D 21
#define N 600010
#define M 1<<D
#define ll long long
using namespace std;
int dep[N],fa[N];
int head[N],to[N<<1],nxt[N<<1],cnt;
int tag[D][M],w[N];
void add(int u,int v)
{
	nxt[++cnt]=head[u];
	to[cnt]=v;
	head[u]=cnt;
}
ll ans;
int dfs(int u)
{
	ll res=w[u];
	dep[u]=dep[fa[u]]+1;
	for(int i=0;i<D;i++) tag[i][(w[u]+dep[u])&((1<<i)-1)]^=1<<i;
	for(int i=0;i<D;i++) res^=tag[i][dep[u]&((1<<i)-1)];
	for(int i=head[u];i;i=nxt[i])
	{
		int v=to[i];
		res^=dfs(v);
	}
	for(int i=0;i<D;i++) res^=tag[i][dep[u]&((1<<i)-1)];
	ans+=res;
	return res;
}
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&w[i]);
	for(int i=2;i<=n;i++) scanf("%d",&fa[i]),add(fa[i],i);
	dep[0]=-1;
	dfs(1);
	printf("%lld\n",ans);
	return 0;
}

```
